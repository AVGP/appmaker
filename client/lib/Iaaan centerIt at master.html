2012-12-02 first 		1 /**
2 * Extra light plugin that centers content relative to
3 * containing element or window, even when overflowing.
4 *
5 * Main purpose was to solve issues with centering content within
6 * dynamic width containers even when content is overflowing.
7 *
8 * Usage:
9 * $( target to be centered ).centerIt({ options });
10 *
11 * @author ian.ainley@gmail
12 *
13 * @param centerVertical Boolean center it vertically?
14 * @param centerHorizontal Boolean center it horizontally?
15 * @param verticalOffset Int offset in px as Int.
16 * @param horizontalOffset
17 * @param setWidth String CSS value for overriding automatic detection of target's width
18 * @param setHeight String same as setWidth for height
19 * @param outerBox Boolean include padding in target's dimensions?
20 * @param setPosition String CSS value for position rule set on target.
21 * @param relativeTo String sets target's position relative to 'parent' or 'window'.
22 * @param bindResize Boolean re-center target on window resize?
23 * can be turned off for containers with static widths.
24 */
25
26 ;(function($){
27
28 var CenterIt = function (target, options) {
29 var s = this, $w = $(window);
30
31 s.targetWidth = 0;
32 s.targetHeight = 0;
33 s.leftDistance = 0;
34 s.topDistance = 0;
35
36 s.options = {
37 centerVertical: true, // Boolean - center it vertically?
38 centerHorizontal: true, // Boolean - center it horizontally?
39 verticalOffset: 0, // Int - offset in px as int.
40 horizontalOffset: 0,
41 setWidth: "auto", // String - CSS value for desired full width of target.
42 setHeight: "auto",
43 outerBox: true, // Boolean - include padding in target dimensions?
44 setPosition: "relative", // String - CSS value for position rule.
45 relativeTo: "parent", // String - dimensions to be relative to "window" or target's "parent".
46 bindResize: true // Boolean - re-center on window resize? not necessary for static width containers.
47 };
48
49 s.centerIt = function () {
50 var css = {};
51
52 css.position = s.o.setPosition;
53
54 if (s.o.centerHorizontal) {
55 css.left = calculateWidth();
56 }
57
58 if (s.o.centerVertical) {
59 css.top = calculateHeight();
60 }
61
62 target.css(css);
63 };
64
65 var calculateWidth = function () {
66 var fullDist = 0;
67
68 if (s.o.setWidth === "auto") {
69 s.targetWidth = s.o.outerBox ? target.outerWidth() : target.width();
70 } else {
71 s.targetWidth = s.o.setWidth;
72 }
73
74 if (s.o.relativeTo === "window") {
75 var fullDist = windowWidth() - (s.targetWidth - s.o.horizontalOffset);
76 } else if (s.o.relativeTo === "parent") {
77 var fullDist = parentWidth() - (s.targetWidth - s.o.horizontalOffset);
78 } else {
79 console.log("Param 'relativeTo' not set correctly. Choose 'parent' or 'window'.");
80
81 return;
82 }
83
84 return s.leftDistance = Math.round(fullDist / 2);
85 };
86
87 var calculateHeight = function () {
88 var fullDist = 0;
89
90 if (s.o.setHeight === "auto") {
91 s.targetHeight = s.o.outerBox ? target.outerHeight() : target.height();
92 } else {
93 s.targetHeight = s.o.setHeight;
94 }
95
96 if (s.o.relativeTo === "window") {
97 fullDist = windowHeight() - (s.targetHeight - s.o.verticalOffset);
98 } else if (s.o.relativeTo === "parent") {
99 fullDist = parentHeight() - (s.targetHeight - s.o.verticalOffset);
100 } else {
101 console.log("Param 'relativeTo' not set correctly. Choose 'parent' or 'window'.");
102
103 return;
104 }
105
106 return s.topDistance = Math.round(fullDist / 2);
107 };
108
109 var windowWidth = function () {
110 return $w.width();
111 };
112
113 var parentWidth = function () {
114 return target.parent().width();;
115 };
116
117 var windowHeight = function () {
118 return $w.height();
119 };
120
121 var parentHeight = function () {
122 return target.parent().height();;
123 };
124
125 s.destroy = function () {
126 $w.unbind('resize.centerit');
127 // allows the plugin to be reinstantiated on the same object after it's wiped.
128 target.removeData('centerit');
129 };
130
131 // s.o stores all params for this instance.
132 s.o = $.extend({}, s.options, options);
133
134 if (s.o.bindResize) {
135 $w.on('resize.centerit', function () {
136 s.centerIt();
137 });
138 }
139
140 s.centerIt();
141 };
142
143 $.fn.centerIt = function (options) {
144 if (this.length < 1) return;
145
146 return this.each(function () {
147 var target = $(this);
148
149 // prevents double instantiation.
150 if (target.data('centerit')) return;
151
152 var centerIt = new CenterIt(target, options);
153
154 // stores this instance in the target object's 'data'.
155 target.data('centerit', centerIt);
156 });
157 };
158
159 })(jQuery);
